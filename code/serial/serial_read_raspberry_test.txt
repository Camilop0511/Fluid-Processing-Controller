$serialPort = fopen('/dev/ttyS0', 'r');
system('stty -F /dev/ttyS0 57600');

$a3_received = false;
while (true) {
    $byte = fread($serialPort, 1);
    if ($byte === "\xa3") {
        $a3_received = true;
    } elseif ($a3_received) {
        echo "Step: " . ord($byte) . "\n";
        $a3_received = false;
    }
}

fclose($serialPort);
